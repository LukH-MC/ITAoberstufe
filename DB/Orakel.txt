SELECT d.department_id AS Abteilungsnummer, d.department_name AS Abteilung, SUM(e.salary) AS Gehaltssumme
FROM departments d
JOIN employees e
ON d.department_id = e.department_id
GROUP BY d.department_id, d.department_name
